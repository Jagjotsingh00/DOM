/* --- TASK 11: QUIZ APP --- */
const questions = [
    { text: "What is 2 + 2?", answer: "4", options: ["2", "4", "5", "6"] },
    { text: "Capital of India?", answer: "Delhi", options: ["Mumbai", "Delhi", "Goa", "Pune"] },
    { text: "Best JS Library?", answer: "React", options: ["React", "Java", "Python", "C++"] }
];

*****************************************
let currentQIndex = 0;
let score = 0;

// HTML Elements
const questionText = document.getElementById("question-text");
const answerBtns = document.querySelectorAll(".answer-btn");
const scoreText = document.getElementById("quiz-score");

function loadQuestion() {
    const currentQ = questions[currentQIndex];
    questionText.innerText = currentQ.text;

    // Update buttons
    answerBtns.forEach((btn, index) => {
        btn.innerText = currentQ.options[index];
        
        // Remove old event listeners (Clone node trick to clear old clicks)
        const newBtn = btn.cloneNode(true);
        btn.parentNode.replaceChild(newBtn, btn);
        
        // Add new click listener
        newBtn.addEventListener("click", () => checkAnswer(currentQ.options[index]));
    });
}

function checkAnswer(selectedOption) {
    const correctOption = questions[currentQIndex].answer;
    
    if (selectedOption === correctOption) {
        alert("Correct! :white_check_mark:");
        score++;
    } else {
        alert("Wrong! :x: Answer was: " + correctOption);
    }

    scoreText.innerText = "Score: " + score;
    currentQIndex++; // Move to next question

    // Check if quiz is over
    if (currentQIndex < questions.length) {
        loadQuestion();
    } else {
        alert("Quiz Over! Final Score: " + score);
        currentQIndex = 0; // Reset
        score = 0;
        loadQuestion();
    }
}

// Start the quiz
loadQuestion();

Push an object { name: name, price: price } into the cart array.

/* --- TASK 12: SHOPPING CART --- */
let cart = []; // Array to store items

function addToCart(name, price) {
    // 1. Add object to array
    cart.push({ name: name, price: price });
    
    // 2. Update the UI
    renderCart();
}

function renderCart() {
    const cartList = document.getElementById("cart-list");
    const totalSpan = document.getElementById("total-price");
    
    // Clear list to avoid duplicates
    cartList.innerHTML = "";
    
    let total = 0;

    cart.forEach((item) => {
        // Create List Item
        const li = document.createElement("li");
        li.innerText = `${item.name} - ‚Çπ${item.price}`;
        cartList.appendChild(li);
        
        // Add to total
        total += item.price;
    });

    // Update Total Text
    totalSpan.innerText = total;
}

const contents = document.querySelectorAll(".content-box");

14. The Task Filter (Array Filtering)
‚Ä¢ Data: let tasks = [{ text: "Learn JS", completed: false }];
‚Ä¢ Logic:
1. Add: New tasks are pushed to the array with completed: false.
2. Toggle: Clicking a task toggles its completed status (true/false) and adds a CSS line-through style.

Filter:
"Show Active" button: Uses tasks.filter(t => !t.completed) and re-renders the list.
"Show Completed" button: Uses tasks.filter(t => t.completed).
"Show All": Renders the whole array.

**********************************************************

/* --- TASK 14: TASK FILTER --- */
let tasks = [];

const taskInput = document.getElementById("task-input");
const addTaskBtn = document.getElementById("add-task-btn");
const taskListUl = document.getElementById("filter-task-list");

// Add Task
addTaskBtn.addEventListener("click", () => {
    if(taskInput.value === "") return;
    
    tasks.push({ text: taskInput.value, completed: false });
    taskInput.value = "";
    renderTasks("all"); // Show all by default
});

// Render Function (The Brain)
function renderTasks(filterType) {
    taskListUl.innerHTML = ""; // Clear current list

    // Filter the array based on button clicked
    let filteredTasks = tasks;
    if (filterType === "active") {
        filteredTasks = tasks.filter(t => !t.completed);
    } else if (filterType === "completed") {
        filteredTasks = tasks.filter(t => t.completed);
    }

    // Loop and Display
    filteredTasks.forEach((task, index) => {
        const li = document.createElement("li");
        li.innerText = task.text;
        
        // If completed, cross it out
        if (task.completed) {
            li.style.textDecoration = "line-through";
        }

        // Toggle completion on click
        li.addEventListener("click", () => {
            // Find the original task in the main array and flip it
            // (Using reference or finding by text/index)
            task.completed = !task.completed; 
            renderTasks(filterType); // Re-render to show changes
        });

        taskListUl.appendChild(li);
    });
}

// Button Listeners
document.getElementById("show-all-btn").addEventListener("click", () => renderTasks("all"));
document.getElementById("show-active-btn").addEventListener("click", () => renderTasks("active"));
document.getElementById("show-completed-btn").addEventListener("click", () => renderTasks("completed"));

15. The Live Form Validator (Events & Disabled Button)
‚Ä¢ Goal: The "Sign Up" button should stay Disabled (greyed out) until the form is valid.
‚Ä¢ Logic:

Add input event listeners to all 3 boxes.
Username Check: If length < 5, show #user-error.
Password Check: If length < 8, show #pass-error.
Match Check: If pass value !== confirm-pass value, show #match-error.
Master Check: If ALL 3 are valid, set submitBtn.disabled = false. Otherwise, keep it true.


***************************************************************************

/* --- TASK 15: LIVE VALIDATOR --- */
const userField = document.getElementById("username");
const passField = document.getElementById("pass");
const confirmField = document.getElementById("confirm-pass");
const submitBtn = document.getElementById("submit-btn");

// Errors
const userErr = document.getElementById("user-error");
const passErr = document.getElementById("pass-error");
const matchErr = document.getElementById("match-error");

function validateForm() {
    let isValid = true;

    // 1. Check Username
    if (userField.value.length < 5) {
        userErr.style.display = "inline";
        isValid = false;
    } else {
        userErr.style.display = "none";
    }

    // 2. Check Password Length
    if (passField.value.length < 8) {
        passErr.style.display = "inline";
        isValid = false;
    } else {
        passErr.style.display = "none";
    }

    // 3. Check Match
    if (passField.value !== confirmField.value) {
        matchErr.style.display = "inline";
        isValid = false;
    } else {
        matchErr.style.display = "none";
    }

    // 4. Enable/Disable Button
    submitBtn.disabled = !isValid; // If valid is true, disabled is false
}

// Attach listeners to ALL inputs
userField.addEventListener("input", validateForm);
passField.addEventListener("input", validateForm);
confirmField.addEventListener("input", validateForm);
